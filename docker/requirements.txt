# Phase 1 — text bot dependencies. Iris's own runtime deps (mcp, httpx, etc.)
# come from pyproject.toml via `pip install -e .` in the Dockerfile.

# Voice extras (pulls PyNaCl + davey) — required from discord.py 2.5+ for
# any voice channel connection, since Discord enforced AEAD encryption +
# the DAVE end-to-end-encryption protocol. Without the [voice] extras the
# bot crashes with "davey library needed in order to use voice" on the
# first `channel.connect()` call. Adds ~5 MB to the container image
# (libsodium binding + DAVE protocol code).
discord.py[voice]>=2.4
# Voice receive — adds VoiceRecvClient + AudioSink to discord.py. Lets Iris
# hear users (Phase 2.2.1). Without this, voice channels are send-only.
# The library is import-gated in docker/voice_listen.py, so the bot still
# boots if you skip it; it just silently runs send-only.
discord-ext-voice-recv>=0.5
claude-agent-sdk>=0.0.18
# Python tzdata fallback. The system `tzdata` apt package SHOULD make
# zoneinfo work on Linux, but if its layout differs from what zoneinfo
# expects, ZoneInfo("Europe/Zurich") silently returns nothing and we fall
# back to UTC — wrecking morning briefings, event pings, and the [Now:] tag.
# This PyPI package ships the IANA tz database in zoneinfo's canonical
# format, so it's always findable regardless of the host distro.
tzdata>=2025.1
